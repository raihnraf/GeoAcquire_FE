    ```typescript
    }
    ```

    Implementation requirements:
    1. Import useForm from 'react-hook-form', zodResolver from '@hookform/resolvers/zod', parcelSchema and ParcelFormData from '@/lib/zod'
    2. Use useForm with:
       - resolver: zodResolver(parcelSchema)
       - defaultValues: { owner_name: '', status: 'free', price_per_sqm: undefined, geometry: { type: 'Polygon', coordinates: [[]] } }
       - mode: 'onTouched' (validate on blur for better UX)
    3. Extract register, handleSubmit, formState: { errors, isSubmitting }, setError from useForm
    4. Handle form submission with try/catch to map server errors to form fields:
       ```typescript
       const handleSubmitForm = async (data: ParcelFormData) => {
         try {
           await onSubmit(data)
         } catch (error: any) {
           if (error.response?.data?.errors) {
             Object.entries(error.response.data.errors).forEach(([field, message]) => {
               setError(field as keyof ParcelFormData, { message: message as string })
             })
           }
         }
       }
       ```
    5. Form fields using FormField component:
       - owner_name: text input, required
       - status: select dropdown (not FormField - needs custom select), options: free, negotiating, target
       - price_per_sqm: number input with valueAsNumber, optional
    6. Status select with color-coded dots:

       ```typescript
       <select {...register('status')} className="...">

         <option value="negotiating">Negotiating</option>
         <option value="target">Target</option>
       </select>
       ```
    7. Form actions at bottom with flex layout (gap-2 mt-6):
       - Cancel button: flex-1 bg-slate-100 text-slate-700 hover:bg-slate-200 py-2
       - Submit button: flex-1 bg-blue-500 text-white hover:bg-blue-600 py-2, disabled when isSubmitting
    8. Show "Saving..." or "Creating..." text in submit button when isSubmitting
